function

iml::swap

概要

iml::swapは引数のそれぞれの値を入れ替える。 


namespace iml {
	template <class T>
	void swap(T& t1, T& t2);
	template <iml::imsize_t N, class T>
	void swap(T(&t1)[N], T(&t2)[N]);
}

サンプルコード

入力


#include <IMathLib/IMathLib.hpp>

int main() {
	double temp[2] = { 5,10 };
	iml::swap(temp[0], temp[1]);
	std::cout << temp[0] << ',' << temp[1] << std::endl;
	return 0;
}

出力


10,5
