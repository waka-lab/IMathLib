function

iml::this_return

概要

iml::this_returnは引数に渡された値をそのまま返す関数オブジェクトである。 


namespace iml {
	template <class T>
	T this_return(const T& v);
}

サンプルコード

入力


#include <IMathLib/IMathLib.hpp>

//fによる補正を与えて和を返す
template <class T, class F = T(*)(const T&)>
T sum(const T& v1, const T& v2, F f = iml::this_return<T>) {
	return f(v1) + f(v2);
}

int main() {

	std::cout << sum(1., 10.) << std::endl;
	std::cout << sum(1., 10., [](double v) {return (v < 5) ? v * 2 : v / 2; }) << std::endl;

	return 0;
}

出力


11
7
