function

iml::move

概要

iml::moveは左辺値を右辺値へとキャストし、ムーブセマンティクスを適用させる。 


namespace iml {
	template <class T>
	constexpr typename remove_reference<T>::type&& move(T&& arg) noexcept;
}

サンプルコード

入力


#include <IMathLib/IMathLib.hpp>

struct test {
	constexpr test() {}
	test(const test&) {
		std::cout << "copy" << std::endl;
	}
	test(test&&){
		std::cout << "move" << std::endl;
	}
};

int main() {
	test temp;
	test temp2(temp);
	test temp3(iml::move(temp));
	return 0;
}

出力


copy
move
