function

iml::forward

概要

iml::forwardは関数の引数TをT&&にキャストして返し、関数テンプレートの引数を転送する。 


namespace iml {
	template<class T>
	constexpr T&& forward(typename remove_reference<T>::type& arg) noexcept;
	template<class T>
	constexpr T&& forward(typename remove_reference<T>::type&& arg) noexcept;
}

サンプルコード

入力


#include <IMathLib/IMathLib.hpp>

template <class... Types>
struct test {
	constexpr test() {}
	template <class F>
	auto operator()(F f, Types... args) {
		return f(iml::forward<Types>(args)...);
	}
};

int main() {
	test<int, int, int> temp;
	std::cout << temp([](int a, int b, int c) {return a + b + c; }, 1, 2, 3) << std::endl;
	return 0;
}

出力


6
