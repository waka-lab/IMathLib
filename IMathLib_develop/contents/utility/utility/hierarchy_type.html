struct

iml::hierarchy_type

概要

iml::hierarchy_typeはテンプレート型Tの階層の中にテンプレート型Sを存在するならばtype=Sとなり、存在しないならばtype=iml::object_type<T>::typeとなる。 


namespace iml {
	template <class T, class S>
	struct hierarchy_type;
}

サンプルコード

入力


#include <IMathLib/IMathLib.hpp>

int main() {
	iml::hierarchy_type<iml::vector<iml::complex<double>, 5>, iml::complex<double>>::type
		temp(0, 1);									//iml::complex<double>
	iml::hierarchy_type<iml::vector<iml::complex<double>, 5>, int>::type
		temp2 = 10;									//double
	std::cout << temp << ',' << temp2 << std::endl;
	return 0;
}

出力


10
