struct

iml::placeholder

概要

iml::placeholderは任意の整数型テンプレート引数として渡すためにusingで定義された構造体である。 


namespace iml {
	template <iml::imsize_t N>
	using placeholder = iml::integral_constant <iml::imsize_t, N>;
}

サンプルコード

入力


#include <IMathLib/IMathLib.hpp>

template <iml::imsize_t N>
void test(iml::placeholder<N>) {
	std::cout << N << std::endl;
}

int main() {
	test(iml::placeholder<10>());
	return 0;
}

出力


10
